Name Property Example (VC++)

This example creates a TableDef object and then names it Parts ID.

CdbDBEngine   DBEng;
CdbDatabase   dbsNorthwind;
CdbQueryDef   qdfCustomers;
CdbRecordset   rstCustomers;
CdbTableDef   tdfPartsID; 

dbsNorthwind =  DBEng.OpenDatabase(_T("Northwind.mdb"));
tdfPartsID = dbsNorthwind.CreateTableDef();
tdfPartsID.SetName(_T("Parts ID"));
//...   ' Create fields.
dbsNorthwind.TableDefs.Append(tdfPartsID);
