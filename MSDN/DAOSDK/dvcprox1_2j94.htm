LoginTimeout Property Example (VC++)

This example sets the LoginTimeout property to 120 seconds and then creates and runs a query on a database on an ODBC server.

CdbDBEngine   DBEng;
CdbDatabase   dbsNorthwind;
CdbQueryDef   qdfCustomers;
CdbRecordset   rstCustomers;

DBEng.SetLoginTimeout(120);      // Timeout in 2 minutes.
dbsNorthwind =  DBEng. OpenDatabase(_T("Northwind.mdb"));
                           // Create query.
qdfCustomers = dbsNorthwind.CreateQueryDef(_T("All Cust"),
_T("SELECT * FROM Customers;"));
qdfCustomers.SetConnect(_T("ODBC;DSN=Human Resources;DATABASE=HRSRVR; UID=Smith; PWD=Sesame"));
                           // Log in to server and run query.
rstCustomers = qdfCustomers.OpenRecordset();
