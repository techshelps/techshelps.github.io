sendto

#include <wsvv.h>

int PASCAL FAR sendto(s, buf, len, flags, to, tolen)
SOCKET s; /* connected socket on which to send */
const char FAR * buf; /* buffer containing data to transmit */
int len; /* length of data in buffer */
int flags; /* specifies optional behavior of the call */
const struct sockaddr FAR * to; /* address structure for destination socket */
int tolen; /* length of address structure */

Send data to a specific destination.

Addendum

The sendto function specifies the protocol ID for the socket using the modified sockaddr structure.
