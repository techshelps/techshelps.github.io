recvfrom

#include <wsvv.h>

int PASCAL FAR recvfrom(s, buf, len, flags, from, fromlen)
SOCKET s; /* bound socket to receive data */
char FAR * buf; /* buffer to receive incoming data */
int len; /* length of buffer */
int flags; /* specifies optional behavior of the call */
struct sockaddr FAR * from; /* address structure to receive source address */
int FAR * fromlen; /* length of address structure */

Receive a datagram and store the source address.

Addendum

The recvfrom function receives a protocol ID for a socket using the modified sockaddr structure.
