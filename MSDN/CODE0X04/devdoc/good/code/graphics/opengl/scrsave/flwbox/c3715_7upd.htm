// Minimum and maximum number of side subdivisions 
#define MINSUBDIV 2 
#define MAXSUBDIV 10

#define MAXPTS (MAXSIDES*MAXSPTS) 
#define MAXSFACES (MAXSUBDIV*MAXSUBDIV) 
#define MAXFACES (MAXSIDES*MAXSFACES) 
#define MAXFPTS 4 
 
// Number of colors used in checkerboarding 
#define NCCOLS 2

typedef GLfloat FLT;

{ 
    FLT x, y, z; 
} PT3;

{ 
    BOOL smooth_colors; 
    BOOL triangle_colors; 
    BOOL cycle_colors; 
    BOOL spin; 
    BOOL bloom; 
    int subdiv; 
    int color_pick; 
    int image_size; 
    int geom; 
    int two_sided; 
} CONFIG;

#define V3Sub(a, b, r) \ 
    ((r)->x = (a)->x-(b)->x, (r)->y = (a)->y-(b)->y, (r)->z = (a)->z-(b)->z) 
#define V3Add(a, b, r) \ 
    ((r)->x = (a)->x+(b)->x, (r)->y = (a)->y+(b)->y, (r)->z = (a)->z+(b)->z) 
#define V3Cross(a, b, r) \ 
    ((r)->x = (a)->y*(b)->z-(b)->y*(a)->z,\ 
     (r)->y = (a)->z*(b)->x-(b)->z*(a)->x,\ 
     (r)->z = (a)->x*(b)->y-(b)->x*(a)->y)
