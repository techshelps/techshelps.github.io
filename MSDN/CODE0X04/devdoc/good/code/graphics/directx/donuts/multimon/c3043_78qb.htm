int randInt( int low, int high ) 
{ 
    int range = high - low; 
    int num = rand() % range; 
    return( num + low ); 
} 
 
double randDouble( double low, double high ) 
{ 
    double range = high - low; 
    double num = range * (double)rand()/(double)RAND_MAX; 
    return( num + low ); 
} 
 
 
int getint(char**p, int def) 
{ 
    int i=0; 
 
 
    while (IS_SPACE(**p)) 
        (*p)++; 
 
    if (!IS_NUM(**p)) 
        return def; 
 
    while (IS_NUM(**p)) 
        i = i*10 + *(*p)++ - '0'; 
 
    while (IS_SPACE(**p)) 
        (*p)++; 
 
    return i; 
}
