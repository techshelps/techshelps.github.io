<% @Language=JScript %> 
 
 
<Script Language=JScript RunAt=Server> 
 
// Define Server Side Script Function 
 
function PrintOutMsg(msg, count) 
{ 
// Output Message count times 
 
for(i = 0; i < count; i++) 
{ 
Response.Write(msg + "<BR>"); 
} 
 
 
// Return number of iterations 
 
return count; 
} 
 
</Script>
