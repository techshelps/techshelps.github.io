#include <windows.h> 
 
 
/* 
 * LibMain(32) 
 * 
 * Purpose: 
 *  Entry point for Win32 DLLs 
 */ 
 
BOOL WINAPI LibMain32(HINSTANCE hInstance, ULONG ulReason 
    , LPVOID pvReserved) 
    { 
    if (DLL_PROCESS_DETACH==ulReason) 
        { 
        return TRUE; 
        } 
    else 
        { 
        if (DLL_PROCESS_ATTACH!=ulReason) 
            return TRUE; 
        } 
 
    return TRUE; 
    }
