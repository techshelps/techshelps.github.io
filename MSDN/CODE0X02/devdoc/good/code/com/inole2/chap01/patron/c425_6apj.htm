/* 
 * CPage::CPage 
 * CPage::~CPage 
 * 
 * Constructor Parameters: 
 *  dwID            DWORD identifier for this page. 
 */ 
 
CPage::CPage(DWORD dwID) 
    { 
    m_dwID=dwID; 
    return; 
    } 
 
CPage::~CPage(void) 
    { 
    return; 
    } 
 
 
 
/* 
 * CPage::GetID 
 * 
 * Return Value: 
 *  DWORD           dwID field in this page. 
 */ 
 
DWORD CPage::GetID(void) 
    { 
    return m_dwID; 
    }
