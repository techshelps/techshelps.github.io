{ 
    RPC_STATUS status;

    status = RpcMgmtStopServerListening(NULL); 
    printf("RpcMgmtStopServerListening returned: 0x%x\n", status); 
    if (status) { 
        exit(status); 
    }

    status = RpcServerUnregisterIf(NULL, NULL, FALSE); 
    printf("RpcServerUnregisterIf returned 0x%x\n", status); 
    if (status) { 
        exit(status); 
    } 
}
