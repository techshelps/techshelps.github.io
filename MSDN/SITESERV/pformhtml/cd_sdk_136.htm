Name Property

This property specifies the name of the route.

Syntax

ReplicationRoute.Name[=Name]

Parameters

Name

The name of the route.

Example

The following example retrieves and displays the names of all routes. 

Option Explicit 
On Error Resume Next 

const CRS_ERROR_NO_MORE_ITEMS  = 0&80003B17

dim ReplServer
set ReplServer = CreateObject("CrsApi.ReplicationServer")
ReplServer.Initialize("")

dim Iterator
Iterator = 0
dim ReplRoute

Wscript.Echo "The routes are:"

do while True
  'Clear any error text
  Err.Clear
  'Get a route
  set ReplRoute = ReplServer.EnumRoutes(Iterator)

  'Quit if empty object returned (no more instances)
  if IsEmpty(ReplRoute) then exit do

  'Quit if "No more items" error 
  dim ReplError
  ReplError = Err.Number
  if ReplError = CRS_ERROR_NO_MORE_ITEMS then exit do

  Wscript.Echo ReplRoute.Name

Loop
'Release objects
set ReplRoute  = Nothing
set ReplServer = Nothing
