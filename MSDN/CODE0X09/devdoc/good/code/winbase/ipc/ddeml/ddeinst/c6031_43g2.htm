/******************************************************************** 
 
   AboutBoxWndProc 
 
   Function to process messages to the about box. 
 
********************************************************************/ 
 
LRESULT APIENTRY AboutBoxWndProc (HWND hwnd, UINT uiMsg, WPARAM wParam, 
      LPARAM lParam) { 
   switch (uiMsg) { 
      case WM_COMMAND: { 
         EndDialog (hwnd, TRUE); 
         return (TRUE); 
      }/*endCase*/ 
   }/*endSwitch*/ 
   return (FALSE); 
}/* end AboutBoxWndProc */ 
 
/******************************************************************** 
 
   ProgressDlgWndProc 
 
   Function to process messages to the about box. 
 
********************************************************************/ 
 
LRESULT APIENTRY ProgressDlgWndProc (HWND hwnd, UINT uiMsg, WPARAM wParam, 
      LPARAM lParam) { 
   switch (uiMsg) { 
      case WM_DESTROY: { 
         return (TRUE); 
      }/*endCase*/ 
   }/*endSwitch*/ 
   return (FALSE); 
}/* end ProgressDlgWndProc */
