Creating a Status Callback Function

The following example is a simple status callback function. Register this callback by using the capSetCallbackOnStatus macro. 

// StatusCallbackProc: status callback function 
// hWnd:               capture window handle 
// nID:                status code for the current status 
// lpStatusText:       status text string for the current status 
// 
LRESULT PASCAL StatusCallbackProc(HWND hWnd, int nID, 
    LPSTR lpStatusText) 
{ 
    if (!ghWndMain) 
        return FALSE; 
 
    if (nID == 0) {           // Clear old status messages. 
        SetWindowText(ghWndMain, (LPSTR) gachAppName); 
        return (LRESULT) TRUE; 
    } 
    // Show the status ID and status text... 
    wsprintf(gachBuffer, "Status# %d: %s", nID, lpStatusText); 
 
    SetWindowText(ghWndMain, (LPSTR)gachBuffer); 
    return (LRESULT) TRUE; 
}
