CURRENCY DBTYPE_CY
DATETIME DBTYPE_DATE
DOUBLE DBTYPE_R8
GUID DBTYPE_GUID
LONG DBTYPE_I4
LONGBINARY DBTYPE_BYTES
LONGTEXT DBTYPE_STR
SHORT DBTYPE_I2
SINGLE DBTYPE_R4
TEXT (< 255 bytes) DBTYPE_STR
