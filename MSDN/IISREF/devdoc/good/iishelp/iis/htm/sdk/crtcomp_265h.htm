IWriteCookie Interface

The IWriteCookie interface enables your component to add new values and set the attributes of the write-only Cookie collection.

Syntax

public interface asp.IWriteCookie 
    extends com.ms.com.IUnknown
{
           //Methods
    public abstract void putExpires(double);
    public abstract void putSecure(boolean);
    public abstract void putPath(java.lang.String);
    public abstract com.ms.com.IUnknown get_NewEnum();
    public abstract boolean getHasKeys();
    public abstract void putDomain(java.lang.String);
    public abstract void putItem(com.ms.com.Variant, java.lang.String);
}
