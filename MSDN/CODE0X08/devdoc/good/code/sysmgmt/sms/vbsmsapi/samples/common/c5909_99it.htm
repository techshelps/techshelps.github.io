Sub GetPixelSizeInTwips(frm As Form, x%, y%) 
    Dim r As Rect 
 
    GetWindowRect frm.hwnd, r 
    x% = frm.Width / (r.right - r.left) 
    y% = frm.Height / (r.bottom - r.top) 
End Sub
