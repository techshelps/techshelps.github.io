DCICREATEINPUT structure

typedef struct {

DCICMD

dcicmd;

// DCI Command header

DWORD

dwCompression;

// format of surface to be created

DWORD

dwMask[3];

// for nonstandard RGB surfaces

DWORD

dwHeight;

// height of surface to be created

DWORD

dwWidth;

// width of input surface

DWORD

dwDCICaps;

// capabilities of surface wanted

DWORD

dwBitCount

// bit depth of the surface

LPVOID

lpSurface;

// pointer to an associated surface


} DCICREATEINPUT, FAR * LPDCICREATEINPUT;
