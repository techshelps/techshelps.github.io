RGNDATA Structure

typedef struct _RGNDATA {

RGNDATAHEADER

rdh;

// region data header

char

buffer[1];

// array of rectangles


} RGNDATA;

typedef struct _RGNDATAHEADER {

DWORD

dwSize;

// size of structure

DWORD

iType;

// type of region. Will be RDH_RECTANGLES

DWORD

nCount;

// number of rectangles

DWORD

nRgnSize;

// size of buffer -- can be zero

RECTL

rcBound;

// bounding rectangle for region


} RGNDATAHEADER;
