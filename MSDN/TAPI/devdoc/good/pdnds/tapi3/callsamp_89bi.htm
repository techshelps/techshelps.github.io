{
    IID iidTermType;

    // convert bstr to iid

                  &iidTermType);

    // see if it's the correct type
    if (IIDTermType == CLSID_FileTerminal)
   {
        //create the terminal
        pTerminalSupport->CreateTerminal(TermType,
                                         &pFileTerminal);
    }
}
