BOOL FAR PASCAL AboutDlg(hDlg, message, wParam, lParam) 
HWND hDlg; 
unsigned message; 
WPARAM wParam; 
LPARAM lParam; 
{ 
    switch (message) { 
        case WM_INITDIALOG: 
            return (TRUE); 
 
        case WM_COMMAND: 
            if (wParam == IDOK 
                || wParam == IDCANCEL) { 
                EndDialog(hDlg, TRUE); 
                return (TRUE); 
            } 
            break; 
    } 
    return (FALSE); 
}
