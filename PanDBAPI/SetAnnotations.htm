SetAnnotations

It saves the annotations applied on the document .

Public Function SetAnnotations(ByVal DocId As Long, ByVal PageNo As Integer, ByVal VersionNo As Single, ByVal NoOfAnnotationGroups As Long, AnnotationGroupList() As Byte, AnnotationBuffer() As String, ByVal sDeleteOtherGroups As String, ByVal FileName As String) As Long

 

Parameters

In/out  

Name

Data type

Description

In

 

DocId

Long

Index of the Document on which annotations have to be saved

In

PageNo

 

Integer

Page No of of the document on which annotations have to be saved

In

VersionNo

Single

Version No of the document on which annotations have to be saved

In

NoOfAnnotationGroups

Long

Number of annotation groups created

In/Out

AnnotationGroupList

Byte[TagAnnotationGroup]

Annotation Group info in the form of byte array

In/Out

AnnotationBuffer

String

Information regarding the annotation

In

sDeleteOtherGroups

String

Specifies Yes for deleting other annotation group else No.

In

FileName

String

 

Name of the Image file if bitmap stamp or signature is to be added 

 

Description

This function is called in both local and shared cabinet .It is called whenever annotations are saved or changed ,a new group is created or notes are added. Parameter 'FileName' contains the name of  image file if any bitmap stamp or signature is added otherwise it is passed as blank. Parameter 'sDeleteOtherGroups' is passed as ‘N” if notes are added else ‘Y’.

 

Return value

Name

Description

 1

In case of success

-1

Invalid cabinet name

-2  

Local cabinet is read only

-3

Cabinet not connected

-201

Insufficient rights for the current operation

-133

User not logged in

 

Example

    Dim lRetval                         As Long

    Dim lDocId                         As Long

    Dim lPageNo                      As Long

    Dim lNoOfAnnotationGroups As Long    

    Dim bAnnotGrpProp()          As Byte

    Dim sAnnotBuffer()             As String

    Dim sFileName                   As String

    Dim sTempStr                    As String

    Dim lhFile                          As Long

    Dim udtAnnotationGroup     As TagAnnotationGroup

    Dim sDeleteOtherGroups    As String

    Dim gPDb                         As PanDb.pdbCls

    Dim sVersionNo                As Single 


    Set gPDb = New PanDb.pdbCls

 

    lDocId = 8

    lPageNo = 1

    s VersionNo = 1               

    lNoOfAnnotationGroups  = 1

    sDeleteOtherGroups    = "N"

 

   udtAnnotationGroup.Name = "Ann1"

    ReDim bAnnotGrpProp(Len(udtAnnotationGroup)-1)

    Call CopyMemory(bAnnotGrpProp(0), udtAnnotationGroup,Len(udtAnnotationGroup))

    ReDim sAnnotBuffer(0)

    sAnnotBuffer(0) = ""

    sFileName = "C:\Annot.Tmp"

    lhFile = FreeFile()

    Open sFileName For Input Access Read As #lhFile

    Do While Not EOF(lhFile)

        Line Input #lhFile, sTempStr

        sAnnotBuffer(0) = sAnnotBuffer(0) & sTempStr & vbCrLf

    Loop

    Close lhFile

    lRetval = gPDb.SetAnnotations(lDocId, lPageNo, sVersionNo, lNoOfAnnotationGroups ,bAnnotGrpProp,sAnnotBuffer,  sDeleteOtherGroups,sFileName)

    If lRetval <> 1 Then

        MsgBox gPDb.ErrDescription

    Else

        Call CopyMemory(udtAnnotationGroup, bAnnotGrpProp(0),

          Len(udtAnnotationGroup))

        MsgBox udtAnnotationGroup.Index & ";" & udtAnnotationGroup.Name

    End If
