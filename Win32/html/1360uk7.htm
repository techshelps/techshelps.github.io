FD_FLAGS

The FD_FLAGS enumeration type defines the flags used with the dwFlags member of the FILEDESCRIPTOR structure.

typedef enum {

FD_CLSID = 0x0001,

FD_SIZEPOINT = 0x0002,

FD_ATTRIBUTES = 0x0004,

FD_CREATETIME = 0x0008,

FD_ACCESSTIME = 0x0010,

FD_WRITESTIME = 0x0020,

FD_FILESIZE = 0x0040,

FD_LINKUI = 0x8000,

} FD_FLAGS;
